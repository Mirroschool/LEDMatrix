72 unsigned char binary_to_base64(unsigned char v) {
73  // Capital letters - 'A' is ascii 65 and base64 0
74  if(v < 26) return v + 'A';
75 
76  // Lowercase letters - 'a' is ascii 97 and base64 26
77  if(v < 52) return v + 71;
78 
79  // Digits - '0' is ascii 48 and base64 52
80  if(v < 62) return v - 4;
81 
82  // '+' is ascii 43 and base64 62
83  if(v == 62) return '+';
84 
85  // '/' is ascii 47 and base64 63
86  if(v == 63) return '/';
87 
88  return 64;
89 }

91 unsigned char base64_to_binary(unsigned char c) {
92  // Capital letters - 'A' is ascii 65 and base64 0
93  if('A' <= c && c <= 'Z') return c - 'A';
94 
95  // Lowercase letters - 'a' is ascii 97 and base64 26
96  if('a' <= c && c <= 'z') return c - 71;
97 
98  // Digits - '0' is ascii 48 and base64 52
99  if('0' <= c && c <= '9') return c + 4;
100 
101  // '+' is ascii 43 and base64 62
102  if(c == '+') return 62;
103 
104  // '/' is ascii 47 and base64 63
105  if(c == '/') return 63;
106 
107  return 255;
108 }

133  unsigned int full_sets = input_length/3;
134 
135  // While there are still full sets of 24 bits...
136  for(unsigned int i = 0; i < full_sets; ++i) {
137  output[0] = binary_to_base64( input[0] >> 2);
138  output[1] = binary_to_base64((input[0] & 0x03) << 4 | input[1] >> 4);
139  output[2] = binary_to_base64((input[1] & 0x0F) << 2 | input[2] >> 6);
140  output[3] = binary_to_base64( input[2] & 0x3F);
141 
142  input += 3;
143  output += 4;
144  }
145 
146  switch(input_length % 3) {
147  case 0:
148  output[0] = '\0';
149  break;
150  case 1:
151  output[0] = binary_to_base64( input[0] >> 2);
152  output[1] = binary_to_base64((input[0] & 0x03) << 4);
153  output[2] = '=';
154  output[3] = '=';

156  break;
157  case 2:
158  output[0] = binary_to_base64( input[0] >> 2);
159  output[1] = binary_to_base64((input[0] & 0x03) << 4 | input[1] >> 4);
