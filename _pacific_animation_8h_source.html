9  //
10  // The code for this animation is more complicated than other examples, and
11  // while it is "ready to run", and documented in general, it is probably not
12  // the best starting point for learning. Nevertheless, it does illustrate some
13  // useful techniques.
14  //
16  //
17  // In this animation, there are four "layers" of waves of light.
18  //
19  // Each layer moves independently, and each is scaled separately.
20  //
21  // All four wave layers are added together on top of each other, and then
22  // another filter is applied that adds "whitecaps" of brightness where the
23  // waves line up with each other more. Finally, another pass is taken
24  // over the led array to 'deepen' (dim) the blues and greens.
25  //
26  // The speed and scale and motion each layer varies slowly within independent
27  // hand-chosen ranges, which is why the code has a lot of low-speed 'beatsin8' functions
28  // with a lot of oddly specific numeric ranges.
29  //
30  // These three custom blue-green color palettes were inspired by the colors found in
31  // the waters off the southern coast of California, https://goo.gl/maps/QQgd97jjHesHZVxQ7
32  //
33  CRGBPalette16 pacifica_palette_1 =
34  { 0x000507, 0x000409, 0x00030B, 0x00030D, 0x000210, 0x000212, 0x000114, 0x000117,
35  0x000019, 0x00001C, 0x000026, 0x000031, 0x00003B, 0x000046, 0x14554B, 0x28AA50 };
36  CRGBPalette16 pacifica_palette_2 =
37  { 0x000507, 0x000409, 0x00030B, 0x00030D, 0x000210, 0x000212, 0x000114, 0x000117,
38  0x000019, 0x00001C, 0x000026, 0x000031, 0x00003B, 0x000046, 0x0C5F52, 0x19BE5F };
39  CRGBPalette16 pacifica_palette_3 =
40  { 0x000208, 0x00030E, 0x000514, 0x00061A, 0x000820, 0x000927, 0x000B2D, 0x000C33,
41  0x000E39, 0x001040, 0x001450, 0x001860, 0x001C70, 0x002080, 0x1040BF, 0x2060FF };

43  // Add one layer of waves into the led array
44  void pacifica_one_layer( CRGBPalette16& p, uint16_t cistart, uint16_t wavescale, uint8_t bri, uint16_t ioff)
45  {
46  uint16_t ci = cistart;
47  uint16_t waveangle = ioff;
48  uint16_t wavescale_half = (wavescale / 2) + 20;
49  for( uint16_t i = 0; i < NUM_LEDS; i++) {
50  waveangle += 250;
51  uint16_t s16 = sin16( waveangle ) + 32768;
52  uint16_t cs = scale16( s16 , wavescale_half ) + wavescale_half;
53  ci += cs;
54  uint16_t sindex16 = sin16( ci) + 32768;
55  uint8_t sindex8 = scale16( sindex16, 240);
56  CRGB c = ColorFromPalette( p, sindex8, bri, LINEARBLEND);
57  leds[i] += c;
58  }
59  }

61  // Add extra 'white' to areas where the four layers of light have lined up brightly
62  void pacifica_add_whitecaps()
63  {
64  uint8_t basethreshold = beatsin8( 9, 55, 65);
65  uint8_t wave = beat8( 7 );
66 
67  for( uint16_t i = 0; i < NUM_LEDS; i++) {
68  uint8_t threshold = scale8( sin8( wave), 20) + basethreshold;
69  wave += 7;
70  uint8_t l = leds[i].getAverageLight();
71  if( l > threshold) {
72  uint8_t overage = l - threshold;
73  uint8_t overage2 = qadd8( overage, overage);
74  leds[i] += CRGB( overage, overage2, qadd8( overage2, overage2));
75  }
76  }
77  }

89  void pacifica_loop()
90  {
91  // Increment the four "color index start" counters, one for each wave layer.
92  // Each is incremented at a different speed, and the speeds vary over time.
93  static uint16_t sCIStart1, sCIStart2, sCIStart3, sCIStart4;
94  static uint32_t sLastms = 0;
95  uint32_t ms = GET_MILLIS();
96  uint32_t deltams = ms - sLastms;
97  sLastms = ms;
98  uint16_t speedfactor1 = beatsin16(3, 179, 269);
99  uint16_t speedfactor2 = beatsin16(4, 179, 269);
100  uint32_t deltams1 = (deltams * speedfactor1) / 256;
101  uint32_t deltams2 = (deltams * speedfactor2) / 256;
102  uint32_t deltams21 = (deltams1 + deltams2) / 2;
103  sCIStart1 += (deltams1 * beatsin88(1011,10,13));
104  sCIStart2 -= (deltams21 * beatsin88(777,8,11));
105  sCIStart3 -= (deltams1 * beatsin88(501,5,7));
106  sCIStart4 -= (deltams2 * beatsin88(257,4,6));
107 
108  // Clear out the LED array to a dim background blue-green
109  fill_solid( leds, NUM_LEDS, CRGB( 2, 6, 10));
110 
111  // Render each of four layers, with different scales and speeds, that vary over time
112  pacifica_one_layer( pacifica_palette_1, sCIStart1, beatsin16( 3, 11 * 256, 14 * 256), beatsin8( 10, 70, 130), 0-beat16( 301) );
113  pacifica_one_layer( pacifica_palette_2, sCIStart2, beatsin16( 4, 6 * 256, 9 * 256), beatsin8( 17, 40, 80), beat16( 401) );
114  pacifica_one_layer( pacifica_palette_3, sCIStart3, 6 * 256, beatsin8( 9, 10,38), 0-beat16(503));
115  pacifica_one_layer( pacifica_palette_3, sCIStart4, 5 * 256, beatsin8( 8, 10,28), beat16(601));
116 
117  // Add brighter 'whitecaps' where the waves lines up more
118  pacifica_add_whitecaps();
119 
120  // Deepen the blues and greens a bit
121  pacifica_deepen_colors();
122  }
