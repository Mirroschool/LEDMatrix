58  uint32_t isqrt(uint32_t n) {
59  if ( n < 2 ) return n;
60  uint32_t smallCandidate = isqrt(n >> 2) << 1;
61  uint32_t largeCandidate = smallCandidate + 1;
62  return (largeCandidate*largeCandidate > n) ? smallCandidate : largeCandidate;

65  // Set pixels to intensity around flare
66  void glow( int x, int y, int z ) {
67  int b = z * 10 / flaredecay + 1;
68  for ( int i=(y-b); i<(y+b); ++i ) {
69  for ( int j=(x-b); j<(x+b); ++j ) {
70  if ( i >=0 && j >= 0 && i < rows && j < cols ) {
71  int d = ( flaredecay * isqrt((x-j)*(x-j) + (y-i)*(y-i)) + 5 ) / 10;
72  uint8_t n = 0;
73  if ( z > d ) n = z - d;
74  if ( n > pix[i][j] ) { // can only get brighter
